CSS – Пример 1
     .cards__row {
     display: flex;
     flex-wrap: wrap;
}

.cards__column {
     flex: 0 1 33.333%;
     padding: 0 15px;
}

.cards__image {
     padding: 0px 0px 58% 0px;
     margin: 0px 0px 20px 0px;
     overflow: hidden;
     position: relative;
}

.cards__image img {
     width: 100%;
     height: 100%;
     object-fit: cover;
     position: absolute;
     top: 0;
     left: 0;
}

/* object-fit: cover;  заставляет объект полностью поместиться внутрь не меняя размеры */
/* object-fit: fill;  позволяет объекту полностью заполнить содержимое родителя, используя свои размеры */
/* object-fit: contain; заставляет объект полностью поместиться внутрь родителя */
/* object-fit: none; вписываются в размер родителя, масштабируюсь так, чтобы показать как можно больше */
